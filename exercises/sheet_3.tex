\begin{question}
  Es sei $R$ ein kommutativer Ring und $P \ideal R$ ein Primideal.
  Zeigen Sie, dass
  \[
              S
    \defined  R \smallsetminus P
    =         \{ r \in R \suchthat r \notin P \}
  \]
  ein multiplikative Teilmenge von $R$ ist.
\end{question}

\begin{question}

  \]
\end{question}
